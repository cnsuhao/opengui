/*!
	\page BW_Container Container
	
	- widgetGroup: "OpenGUI"
	- widgetType:  "Container"
	
	\section Description
	The container widget is just a simple widget that has very few features. Most of all, it is just an implementation of a blank widget so that you can use it to group several other widgets together.
	
	\see Class Documentation: OpenGUI::Widgets::Container
	
	\section Properties
	- Position : FVector2
	- Size : FVector2
	- Enabled : Boolean


	\section Usage	
	Example XML:
\code
<Widget name="myContainer" group="OpenGUI" type="Container">
	<Property name="position" value="(0.59875 x 0.465)" />
	<Property name="size" value="(0.40125 x 0.535)" />
</Widget>

\endcode

	Example Code:
\code
Widgets::Container* container = 
	static_cast<Widgets::Container*>( System::getSingleton().createWidget("Container","OpenGUI") );
container->setPos( 0.59875f, 0.465f );
container->setSize( 0.40125f, 0.535f );
\endcode
*/

/*!
	\page BW_FrameImage FrameImage
	
	- widgetGroup: "OpenGUI"
	- widgetType:  "FrameImage"
	
	\section Description
	Draws a static image to the screen much like StaticImage, but in such a way that pixel stretching is contained to predefined "acceptable" areas. The drawn image is comprised of 9 zones, each of which is actually defined as independent Imagery. All non-stretched dimensions automatically preserve the original pixel density of the source image when displayed. So if your border is 24 pixels thick in your source image, it will always be displayed at 24 pixels thick, regardless of the current screen resolution.
	
	List of zones and stretch method performed:
	- Upper Left (no stretching)
	- Upper Middle (stretches in X only)
	- Upper Right (no stretching)
	- Middle Left (stretches in Y only)
	- Middle Middle (center area)
	- Middle Right (stretches in Y only)
	- Lower Left (no stretching)
	- Lower Middle (stretches in X only)
	- Lower Right (no stretching)
	
	\image html FrameImage_stretching.jpg
	
	\note This widget determines the "border thickness" for each side independently of the others, and will always use the "thickest" value found as the thickness for that entire side.
	
	
	\see Class Documentation: OpenGUI::Widgets::FrameImage
	
	\section Properties
	- Position : FVector2
	- Size : FVector2
	- Enabled : Boolean
	- Imagery_UL : String
	- Imagery_UM : String
	- Imagery_UR : String
	- Imagery_ML : String
	- Imagery_MM : String
	- Imagery_MR : String	
	- Imagery_LL : String
	- Imagery_LM : String
	- Imagery_LR : String

	\section Usage	
	Example XML:
\code
<Widget name="myFrameImage" group="OpenGUI" type="FrameImage">
	<Property name="Imagery_UL" value="myImageryUL" />
	<Property name="Imagery_UM" value="myImageryUM" />
	<Property name="Imagery_UR" value="myImageryUR" />
	<Property name="Imagery_ML" value="myImageryUL" />
	<Property name="Imagery_MM" value="myImageryUM" />
	<Property name="Imagery_MR" value="myImageryUR" />
	<Property name="Imagery_LL" value="myImageryLL" />
	<Property name="Imagery_LM" value="myImageryLM" />
	<Property name="Imagery_LR" value="myImageryLR" />
	
	<Property name="position" value="(0.59875 x 0.465)" />
	<Property name="size" value="(0.40125 x 0.535)" />
</Widget>
\endcode

	Example Code:
\code
Widgets::FrameImage* frameImage = 
	static_cast<Widgets::FrameImage*>( System::getSingleton().createWidget("FrameImage","OpenGUI") );
frameImage->setImagery_UL("myImageryUL");
frameImage->setImagery_UM("myImageryUM");
frameImage->setImagery_UR("myImageryUR");
frameImage->setImagery_ML("myImageryML");
frameImage->setImagery_MM("myImageryMM");
frameImage->setImagery_MR("myImageryMR");
frameImage->setImagery_LL("myImageryLL");
frameImage->setImagery_LM("myImageryLM");
frameImage->setImagery_LR("myImageryLR");

frameImage->setPos( 0.59875f, 0.465f );
frameImage->setSize( 0.40125f, 0.535f );
\endcode
*/

/*!
	\page BW_TextLabel TextLabel
	
	- widgetGroup: "OpenGUI"
	- widgetType:  "TextLabel"
	
	\section Description
	Provides a full featured method of writing text to the screen. Supports various alignment types, and optional word wrapping.
	
	\see Class Documentation: OpenGUI::Widgets::TextLabel, and OpenGUI::TextAlignment
	
	\section Properties
	- Position : FVector2
	- Size : FVector2
	- Enabled : Boolean
	- Text : String
	- Font : String
	- FontSize : Integer
	- Alignment_H : ENUM{ ALIGN_LEFT, ALIGN_CENTER, ALIGN_RIGHT, ALIGN_JUSTIFIED }
	- Alignment_V : ENUM{ ALIGN_TOP, ALIGN_CENTER, ALIGN_BOTTOM, ALIGN_JUSTIFIED }
	- AutoWrap : Boolean

	\section Usage	
	Example XML:
\code
<Widget name="myTextLabel" group="OpenGUI" type="TextLabel">
	<Property name="Font" value="myFont" />
	<Property name="FontSize" value="16" />
	<Property name="Alignment_H" value="ALIGN_LEFT" />
	<Property name="Alignment_V" value="ALIGN_CENTER" />
	<Property name="AutoWrap" value="FALSE" />
	<Property name="Text">This is my text string!
	&#10;And this is the second line.</Property>
	<Property name="position" value="(0.59875 x 0.465)" />
	<Property name="size" value="(0.40125 x 0.535)" />
</Widget>
\endcode

	Example Code:
\code
Widgets::TextLabel* textLabel = 
	static_cast<Widgets::TextLabel*>( System::getSingleton().createWidget("TextLabel","OpenGUI") );
textLabel->setFont("myFont");
textLabel->setFontSize(16);
textLabel->setAlignment_Horiz(TextAlignment::ALIGN_LEFT);
textLabel->setAlignment_Vert(TextAlignment::ALIGN_CENTER);
textLabel->setAutoWrap(false);
textLabel->setText("This is my text string! \nAnd this is the second line.");
textLabel->setPos( 0.59875f, 0.465f );
textLabel->setSize( 0.40125f, 0.535f );
\endcode
*/
